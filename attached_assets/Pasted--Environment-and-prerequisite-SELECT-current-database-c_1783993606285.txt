-- Environment and prerequisite
SELECT current_database(), current_user, version();
SELECT to_regprocedure('digest(text,text)') AS pgcrypto_digest;

-- Control objects
SELECT to_regclass('content_sources') AS content_sources,
       to_regclass('security_events') AS security_events,
       to_regclass('security_rate_limits') AS security_rate_limits;

SELECT table_name, column_name, is_nullable, column_default
FROM information_schema.columns
WHERE (table_name = 'document_versions' AND column_name IN (
  'lifecycle_state','classification','source_id','source_origin_uri',
  'scan_status','scan_findings','approved_by','retention_until'
))
OR (table_name = 'users' AND column_name = 'max_classification')
OR (table_name = 'sessions' AND column_name IN ('auth_method','auth_time'))
ORDER BY table_name, ordinal_position;

-- No unsafe active version may exist
SELECT count(*) AS unsafe_active_versions
FROM document_versions
WHERE is_active = true
  AND (
    lifecycle_state <> 'active'
    OR approved_at IS NULL
    OR source_id IS NULL
    OR scan_status NOT IN ('clean','legacy_accepted')
  );

-- Governance inventory
SELECT lifecycle_state, scan_status, classification, count(*)
FROM document_versions
GROUP BY lifecycle_state, scan_status, classification
ORDER BY lifecycle_state, scan_status, classification;

SELECT role, max_classification, count(*)
FROM users
GROUP BY role, max_classification
ORDER BY role, max_classification;

SELECT program_id, name, owner_name, approved_at, approval_basis
FROM content_sources
ORDER BY program_id, name;

-- Trigger and constraint evidence
SELECT tgname, tgrelid::regclass AS table_name
FROM pg_trigger
WHERE NOT tgisinternal
  AND tgname IN (
    'security_events_append_only',
    'document_versions_audit_insert',
    'document_versions_audit_lifecycle',
    'content_sources_audit_insert',
    'content_sources_audit_update'
  )
ORDER BY tgname;

SELECT conname, conrelid::regclass AS table_name
FROM pg_constraint
WHERE conname IN (
  'document_versions_lifecycle_check',
  'document_versions_classification_check',
  'document_versions_scan_status_check',
  'document_versions_active_control_check',
  'document_versions_separation_check',
  'documents_lifecycle_check',
  'users_max_classification_check',
  'sessions_auth_method_check'
)
ORDER BY conname;

-- Durable deployment receipt. Run once in the intended environment.
SELECT *
FROM append_security_event(
  'deployment.p0_p1_ddl_verified',
  'success',
  NULL, NULL, 'replit_agent',
  NULL,
  'database',
  current_database(),
  NULL, NULL,
  jsonb_build_object('ddlSha256', '44fd12ae4663552d518615ca7d8939d473b01126ceadaeb8be315a85902e3736')
);

-- Hash-chain linkage check
WITH ordered AS (
  SELECT sequence, previous_hash,
         lag(event_hash) OVER (ORDER BY sequence) AS expected_previous_hash
  FROM security_events
)
SELECT count(*) AS broken_hash_links
FROM ordered
WHERE sequence > (SELECT min(sequence) FROM security_events)
  AND previous_hash IS DISTINCT FROM expected_previous_hash;
