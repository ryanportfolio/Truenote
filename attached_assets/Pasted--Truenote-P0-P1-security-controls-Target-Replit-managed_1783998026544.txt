-- Truenote P0/P1 security controls
-- Target: Replit-managed PostgreSQL (Neon)
-- Apply in a maintenance window BEFORE deploying the matching application code.
-- Idempotent for repeated Replit Agent execution.

BEGIN;

-- Replit/Neon prerequisite used by the audit hash chain. The existing project
-- already uses gen_random_uuid(); this check makes a missing pgcrypto digest
-- fail with a clear message before any schema change is applied.
DO $$
BEGIN
  IF to_regprocedure('digest(text,text)') IS NULL THEN
    RAISE EXCEPTION 'pgcrypto digest(text,text) is required before P0/P1 DDL';
  END IF;
END $$;

-- 1. Approved content-source registry.
CREATE TABLE IF NOT EXISTS content_sources (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  program_id uuid NOT NULL REFERENCES programs(id) ON DELETE RESTRICT,
  name text NOT NULL,
  origin_type text NOT NULL CHECK (
    origin_type IN ('manual_upload', 'sharepoint', 'confluence', 's3', 'other')
  ),
  base_uri text,
  owner_name text NOT NULL,
  is_active boolean NOT NULL DEFAULT true,
  created_by uuid REFERENCES users(id) ON DELETE SET NULL,
  created_at timestamptz NOT NULL DEFAULT now(),
  approved_by uuid REFERENCES users(id) ON DELETE SET NULL,
  approved_at timestamptz,
  approval_basis text,
  retired_at timestamptz,
  CONSTRAINT content_sources_approval_check CHECK (
    (is_active = false) OR (approved_at IS NOT NULL AND owner_name <> '')
  )
);

CREATE UNIQUE INDEX IF NOT EXISTS content_sources_program_name_uidx
  ON content_sources (program_id, lower(name));
CREATE INDEX IF NOT EXISTS content_sources_program_active_idx
  ON content_sources (program_id, is_active, approved_at);

-- Bootstrap one controlled manual source plus one explicit legacy source per
-- existing program. Security/Data Steward must review owner and approval basis.
INSERT INTO content_sources (
  program_id, name, origin_type, owner_name, approved_at, approval_basis
)
SELECT
  p.id,
  'Manual administrator upload',
  'manual_upload',
  'Program data steward',
  now(),
  'P0/P1 bootstrap default; owner and authority require Security review'
FROM programs p
ON CONFLICT DO NOTHING;

INSERT INTO content_sources (
  program_id, name, origin_type, owner_name, approved_at, approval_basis
)
SELECT
  p.id,
  'Legacy migration source',
  'other',
  'Legacy owner review required',
  now(),
  'Existing content migrated into controlled lifecycle; formal provenance review required'
FROM programs p
ON CONFLICT DO NOTHING;

-- 2. Document and version lifecycle, provenance, classification, review,
-- scanner evidence, revocation, and retention.
ALTER TABLE documents
  ADD COLUMN IF NOT EXISTS lifecycle_state text NOT NULL DEFAULT 'active',
  ADD COLUMN IF NOT EXISTS retired_at timestamptz,
  ADD COLUMN IF NOT EXISTS retired_by uuid REFERENCES users(id) ON DELETE SET NULL,
  ADD COLUMN IF NOT EXISTS retirement_reason text;

ALTER TABLE document_versions
  ADD COLUMN IF NOT EXISTS lifecycle_state text NOT NULL DEFAULT 'submitted',
  ADD COLUMN IF NOT EXISTS classification text NOT NULL DEFAULT 'internal',
  ADD COLUMN IF NOT EXISTS source_id uuid REFERENCES content_sources(id) ON DELETE RESTRICT,
  ADD COLUMN IF NOT EXISTS source_origin_uri text,
  ADD COLUMN IF NOT EXISTS source_owner text,
  ADD COLUMN IF NOT EXISTS original_file_name text,
  ADD COLUMN IF NOT EXISTS scan_status text NOT NULL DEFAULT 'pending',
  ADD COLUMN IF NOT EXISTS scan_engine text,
  ADD COLUMN IF NOT EXISTS scan_id text,
  ADD COLUMN IF NOT EXISTS scan_findings jsonb NOT NULL DEFAULT '[]'::jsonb,
  ADD COLUMN IF NOT EXISTS scan_completed_at timestamptz,
  ADD COLUMN IF NOT EXISTS approved_by uuid REFERENCES users(id) ON DELETE SET NULL,
  ADD COLUMN IF NOT EXISTS approved_at timestamptz,
  ADD COLUMN IF NOT EXISTS approval_notes text,
  ADD COLUMN IF NOT EXISTS activated_at timestamptz,
  ADD COLUMN IF NOT EXISTS retired_at timestamptz,
  ADD COLUMN IF NOT EXISTS rejected_by uuid REFERENCES users(id) ON DELETE SET NULL,
  ADD COLUMN IF NOT EXISTS rejected_at timestamptz,
  ADD COLUMN IF NOT EXISTS rejection_reason text,
  ADD COLUMN IF NOT EXISTS revoked_by uuid REFERENCES users(id) ON DELETE SET NULL,
  ADD COLUMN IF NOT EXISTS revoked_at timestamptz,
  ADD COLUMN IF NOT EXISTS revocation_reason text,
  ADD COLUMN IF NOT EXISTS retention_until timestamptz NOT NULL DEFAULT (now() + interval '365 days');

-- Backfill every pre-control version. `legacy_accepted` is deliberately
-- distinguishable from a real clean scanner verdict.
UPDATE document_versions dv
SET
  source_id = COALESCE(
    dv.source_id,
    (
      SELECT cs.id
      FROM content_sources cs
      JOIN documents d ON d.program_id = cs.program_id
      WHERE d.id = dv.document_id
        AND cs.name = 'Legacy migration source'
      LIMIT 1
    )
  ),
  source_origin_uri = COALESCE(
    dv.source_origin_uri,
    'legacy-object-storage://' || COALESCE(dv.source_url, dv.id::text)
  ),
  source_owner = COALESCE(dv.source_owner, 'Legacy owner review required'),
  original_file_name = COALESCE(
    dv.original_file_name,
    nullif(regexp_replace(COALESCE(dv.source_url, ''), '^.*/', ''), ''),
    'legacy-document'
  ),
  lifecycle_state = CASE WHEN dv.is_active THEN 'active' ELSE 'retired' END,
  scan_status = CASE
    WHEN dv.scan_status IN ('clean', 'infected', 'unavailable', 'error') THEN dv.scan_status
    ELSE 'legacy_accepted'
  END,
  scan_completed_at = COALESCE(dv.scan_completed_at, dv.uploaded_at, now()),
  approved_at = CASE
    WHEN dv.is_active THEN COALESCE(dv.approved_at, dv.uploaded_at, now())
    ELSE dv.approved_at
  END,
  approval_notes = CASE
    WHEN dv.is_active THEN COALESCE(
      dv.approval_notes,
      'Legacy active content backfilled during P0/P1 rollout; provenance and scan review required'
    )
    ELSE dv.approval_notes
  END,
  activated_at = CASE
    WHEN dv.is_active THEN COALESCE(dv.activated_at, dv.uploaded_at, now())
    ELSE dv.activated_at
  END,
  retired_at = CASE
    WHEN NOT dv.is_active THEN COALESCE(dv.retired_at, now())
    ELSE dv.retired_at
  END
WHERE
  dv.source_id IS NULL
  OR dv.source_origin_uri IS NULL
  OR dv.source_owner IS NULL
  OR dv.original_file_name IS NULL;

ALTER TABLE document_versions
  ALTER COLUMN source_id SET NOT NULL,
  ALTER COLUMN source_origin_uri SET NOT NULL,
  ALTER COLUMN source_owner SET NOT NULL,
  ALTER COLUMN original_file_name SET NOT NULL;

DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_constraint WHERE conname = 'document_versions_lifecycle_check'
  ) THEN
    ALTER TABLE document_versions ADD CONSTRAINT document_versions_lifecycle_check CHECK (
      lifecycle_state IN (
        'submitted', 'scanning', 'parsing', 'pending_review', 'active',
        'retired', 'quarantined', 'rejected', 'revoked', 'failed'
      )
    );
  END IF;
  IF NOT EXISTS (
    SELECT 1 FROM pg_constraint WHERE conname = 'document_versions_classification_check'
  ) THEN
    ALTER TABLE document_versions ADD CONSTRAINT document_versions_classification_check CHECK (
      classification IN ('public', 'internal', 'confidential', 'restricted')
    );
  END IF;
  IF NOT EXISTS (
    SELECT 1 FROM pg_constraint WHERE conname = 'document_versions_scan_status_check'
  ) THEN
    ALTER TABLE document_versions ADD CONSTRAINT document_versions_scan_status_check CHECK (
      scan_status IN (
        'pending', 'running', 'clean', 'infected', 'unavailable',
        'error', 'legacy_accepted'
      )
    );
  END IF;
  IF NOT EXISTS (
    SELECT 1 FROM pg_constraint WHERE conname = 'document_versions_active_control_check'
  ) THEN
    ALTER TABLE document_versions ADD CONSTRAINT document_versions_active_control_check CHECK (
      is_active = false OR (
        lifecycle_state = 'active'
        AND approved_at IS NOT NULL
        AND source_id IS NOT NULL
        AND scan_status IN ('clean', 'legacy_accepted')
      )
    );
  END IF;
  IF NOT EXISTS (
    SELECT 1 FROM pg_constraint WHERE conname = 'document_versions_separation_check'
  ) THEN
    ALTER TABLE document_versions ADD CONSTRAINT document_versions_separation_check CHECK (
      approved_by IS NULL OR uploaded_by IS NULL OR approved_by::text <> uploaded_by
    );
  END IF;
  IF NOT EXISTS (
    SELECT 1 FROM pg_constraint WHERE conname = 'documents_lifecycle_check'
  ) THEN
    ALTER TABLE documents ADD CONSTRAINT documents_lifecycle_check CHECK (
      lifecycle_state IN ('active', 'retired')
    );
  END IF;
END $$;

CREATE INDEX IF NOT EXISTS document_versions_review_queue_idx
  ON document_versions (lifecycle_state, uploaded_at DESC)
  WHERE lifecycle_state IN ('pending_review', 'quarantined');
CREATE INDEX IF NOT EXISTS document_versions_classification_active_idx
  ON document_versions (classification, document_id)
  WHERE is_active = true AND lifecycle_state = 'active';
CREATE INDEX IF NOT EXISTS document_versions_source_idx
  ON document_versions (source_id);
CREATE INDEX IF NOT EXISTS document_versions_retention_idx
  ON document_versions (retention_until)
  WHERE lifecycle_state IN ('retired', 'rejected', 'revoked');

-- 3. User clearance and session authentication evidence.
CREATE TABLE IF NOT EXISTS security_control_metadata (
  key text PRIMARY KEY,
  applied_at timestamptz NOT NULL DEFAULT now(),
  details jsonb NOT NULL DEFAULT '{}'::jsonb
);

ALTER TABLE users
  ADD COLUMN IF NOT EXISTS max_classification text NOT NULL DEFAULT 'internal';

DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM security_control_metadata WHERE key = 'p0_p1_clearance_bootstrap'
  ) THEN
    UPDATE users
    SET max_classification = CASE role::text
      WHEN 'super_user' THEN 'restricted'
      WHEN 'senior_manager' THEN 'restricted'
      WHEN 'manager' THEN 'confidential'
      ELSE 'internal'
    END;
    INSERT INTO security_control_metadata (key, details)
    VALUES (
      'p0_p1_clearance_bootstrap',
      '{"super_user":"restricted","senior_manager":"restricted","manager":"confidential","csr":"internal"}'::jsonb
    );
  END IF;
END $$;

DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_constraint WHERE conname = 'users_max_classification_check'
  ) THEN
    ALTER TABLE users ADD CONSTRAINT users_max_classification_check CHECK (
      max_classification IN ('public', 'internal', 'confidential', 'restricted')
    );
  END IF;
END $$;

ALTER TABLE sessions
  ADD COLUMN IF NOT EXISTS auth_method text NOT NULL DEFAULT 'local',
  ADD COLUMN IF NOT EXISTS auth_time timestamptz NOT NULL DEFAULT now();

DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_constraint WHERE conname = 'sessions_auth_method_check'
  ) THEN
    ALTER TABLE sessions ADD CONSTRAINT sessions_auth_method_check CHECK (
      auth_method IN ('local', 'oidc')

  END IF;
END $$;

-- 4. Distributed fixed-window rate-limit counters.
CREATE TABLE IF NOT EXISTS security_rate_limits (
  scope text NOT NULL,
  subject text NOT NULL,
  window_start timestamptz NOT NULL,
  request_count integer NOT NULL CHECK (request_count > 0),
  expires_at timestamptz NOT NULL,
  PRIMARY KEY (scope, subject, window_start)
);

CREATE INDEX IF NOT EXISTS security_rate_limits_expiry_idx
  ON security_rate_limits (expires_at);

-- 5. Append-only, hash-chained security events.
CREATE TABLE IF NOT EXISTS security_events (
  sequence bigint GENERATED ALWAYS AS IDENTITY UNIQUE,
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  occurred_at timestamptz NOT NULL DEFAULT now(),
  action text NOT NULL,
  outcome text NOT NULL CHECK (outcome IN ('success', 'denied', 'failure')),
  actor_user_id uuid,
  actor_email text,
  actor_role text,
  program_id uuid,
  resource_type text,
  resource_id text,
  request_id text,
  source_ip text,
  details jsonb NOT NULL DEFAULT '{}'::jsonb,
  previous_hash text,
  event_hash text NOT NULL UNIQUE
);

CREATE INDEX IF NOT EXISTS security_events_occurred_idx
  ON security_events (occurred_at DESC);
CREATE INDEX IF NOT EXISTS security_events_program_occurred_idx
  ON security_events (program_id, occurred_at DESC);
CREATE INDEX IF NOT EXISTS security_events_action_occurred_idx
  ON security_events (action, occurred_at DESC);

CREATE OR REPLACE FUNCTION append_security_event(
  p_action text,
  p_outcome text,
  p_actor_user_id uuid,
  p_actor_email text,
  p_actor_role text,
  p_program_id uuid,
  p_resource_type text,
  p_resource_id text,
  p_request_id text,
  p_source_ip text,
  p_details jsonb
)
RETURNS TABLE(id uuid, occurred_at timestamptz, event_hash text)
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public, pg_catalog
AS $$
DECLARE
  v_id uuid := gen_random_uuid();
  v_occurred_at timestamptz := clock_timestamp();
  v_previous_hash text;
  v_event_hash text;
BEGIN
  IF p_action IS NULL OR p_action = '' THEN
    RAISE EXCEPTION 'security event action is required';
  END IF;
  IF p_outcome NOT IN ('success', 'denied', 'failure') THEN
    RAISE EXCEPTION 'security event outcome is invalid';
  END IF;

  PERFORM pg_advisory_xact_lock(hashtext('truenote.security_events.hash_chain'));
  SELECT se.event_hash
  INTO v_previous_hash
  FROM security_events se
  ORDER BY se.sequence DESC
  LIMIT 1;

  v_event_hash := encode(
    digest(
      concat_ws(
        '|',
        COALESCE(v_previous_hash, ''),
        v_id::text,
        v_occurred_at::text,
        p_action,
        p_outcome,
        COALESCE(p_actor_user_id::text, ''),
        COALESCE(p_actor_email, ''),
        COALESCE(p_actor_role, ''),
        COALESCE(p_program_id::text, ''),
        COALESCE(p_resource_type, ''),
        COALESCE(p_resource_id, ''),
        COALESCE(p_request_id, ''),
        COALESCE(p_source_ip, ''),
        COALESCE(p_details, '{}'::jsonb)::text
      ),
      'sha256'
    ),
    'hex'
  );

  INSERT INTO security_events (
    id, occurred_at, action, outcome, actor_user_id, actor_email, actor_role,
    program_id, resource_type, resource_id, request_id, source_ip, details,
    previous_hash, event_hash
  ) VALUES (
    v_id, v_occurred_at, p_action, p_outcome, p_actor_user_id, p_actor_email,
    p_actor_role, p_program_id, p_resource_type, p_resource_id, p_request_id,
    p_source_ip, COALESCE(p_details, '{}'::jsonb), v_previous_hash, v_event_hash
  );

  RETURN QUERY SELECT v_id, v_occurred_at, v_event_hash;
END;
$$;

CREATE OR REPLACE FUNCTION block_security_event_mutation()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
  RAISE EXCEPTION 'security_events is append-only';
END;
$$;

DROP TRIGGER IF EXISTS security_events_append_only ON security_events;
CREATE TRIGGER security_events_append_only
BEFORE UPDATE OR DELETE ON security_events
FOR EACH ROW EXECUTE FUNCTION block_security_event_mutation();

-- Transactional audit for every document-version lifecycle transition.
CREATE OR REPLACE FUNCTION audit_document_version_lifecycle()
RETURNS trigger
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public, pg_catalog
AS $$
DECLARE
  v_program_id uuid;
  v_actor_id uuid;
  v_actor_email text;
  v_actor_role text;
  v_actor_text text;
  v_old_state text;
BEGIN
  SELECT d.program_id INTO v_program_id
  FROM documents d
  WHERE d.id = NEW.document_id;

  v_old_state := CASE WHEN TG_OP = 'INSERT' THEN NULL ELSE OLD.lifecycle_state END;
  v_actor_text := CASE NEW.lifecycle_state
    WHEN 'active' THEN NEW.approved_by::text
    WHEN 'rejected' THEN NEW.rejected_by::text
    WHEN 'revoked' THEN NEW.revoked_by::text
    ELSE NEW.uploaded_by
  END;
  IF v_actor_text ~* '^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$' THEN
    v_actor_id := v_actor_text::uuid;
    SELECT u.email, u.role::text INTO v_actor_email, v_actor_role
    FROM users u WHERE u.id = v_actor_id;
  END IF;

  PERFORM append_security_event(
    'document.lifecycle.' || NEW.lifecycle_state,
    'success',
    v_actor_id,
    v_actor_email,
    v_actor_role,
    v_program_id,
    'document_version',
    NEW.id::text,
    NULL,
    NULL,
    jsonb_build_object(
      'documentId', NEW.document_id,
      'versionNumber', NEW.version_number,
      'previousState', v_old_state,
      'newState', NEW.lifecycle_state,
      'classification', NEW.classification,
      'scanStatus', NEW.scan_status,
      'isActive', NEW.is_active
    )
  );
  RETURN NEW;
END;
$$;

DROP TRIGGER IF EXISTS document_versions_audit_insert ON document_versions;
CREATE TRIGGER document_versions_audit_insert
AFTER INSERT ON document_versions
FOR EACH ROW EXECUTE FUNCTION audit_document_version_lifecycle();

DROP TRIGGER IF EXISTS document_versions_audit_lifecycle ON document_versions;
CREATE TRIGGER document_versions_audit_lifecycle
AFTER UPDATE OF lifecycle_state ON document_versions
FOR EACH ROW
WHEN (OLD.lifecycle_state IS DISTINCT FROM NEW.lifecycle_state)
EXECUTE FUNCTION audit_document_version_lifecycle();

-- Transactional audit for source-registry creation/retirement.
CREATE OR REPLACE FUNCTION audit_content_source_change()
RETURNS trigger
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public, pg_catalog
AS $$
DECLARE
  v_actor_id uuid;
  v_actor_email text;
  v_actor_role text;
BEGIN
  v_actor_id := COALESCE(NEW.approved_by, NEW.created_by);
  SELECT u.email, u.role::text INTO v_actor_email, v_actor_role
  FROM users u WHERE u.id = v_actor_id;
  PERFORM append_security_event(
    CASE WHEN TG_OP = 'INSERT' THEN 'content_source.created' ELSE 'content_source.changed' END,
    'success',
    v_actor_id,
    v_actor_email,
    v_actor_role,
    NEW.program_id,
    'content_source',
    NEW.id::text,
    NULL,
    NULL,
    jsonb_build_object(
      'name', NEW.name,
      'originType', NEW.origin_type,
      'ownerName', NEW.owner_name,
      'isActive', NEW.is_active,
      'approvedAt', NEW.approved_at
    )
  );
  RETURN NEW;
END;
$$;

DROP TRIGGER IF EXISTS content_sources_audit_insert ON content_sources;
CREATE TRIGGER content_sources_audit_insert
AFTER INSERT ON content_sources
FOR EACH ROW EXECUTE FUNCTION audit_content_source_change();

DROP TRIGGER IF EXISTS content_sources_audit_update ON content_sources;
CREATE TRIGGER content_sources_audit_update
AFTER UPDATE ON content_sources
FOR EACH ROW EXECUTE FUNCTION audit_content_source_change();

COMMIT;
